  276.     private function _parse()
  277.     {
  278.         $pos = 0;
  279.         $code = ord($this->_data[$pos]) | ord($this->_data[$pos + 1]) << 8;
  280.         $length = ord($this->_data[$pos + 2]) | ord($this->_data[$pos + 3]) << 8;
  281.         $version = ord($this->_data[$pos + 4]) | ord($this->_data[$pos + 5]) << 8;
  282.         $substreamType = ord($this->_data[$pos + 6]) | ord($this->_data[$pos + 7]) << 8;
  283.  
  284.  
  285.         if (($version != self::XLS_BIFF8) && ($version != self::XLS_BIFF7)) {
  286.             return false;
  287.         }
  288.         if ($substreamType != self::XLS_WorkbookGlobals){
  289.             return false;
  290.         }
  291.         $pos += $length + 4;
  292.         $code = ord($this->_data[$pos]) | ord($this->_data[$pos + 1]) << 8;
  293.  
  294.         $length = ord($this->_data[$pos + 2]) | ord($this->_data[$pos + 3]) << 8;
  295.         while ($code != self::XLS_Type_EOF){
  296.             switch ($code) {
  297.                 case self::XLS_Type_SST:
  298.                     /**
  299.                      * SST - Shared String Table
  300.                      *
  301.                      * This record contains a list of all strings used anywhere
  302.                      * in the workbook. Each string occurs only once. The
  303.                      * workbook uses indexes into the list to reference the
  304.                      * strings.
  305.                      *
  306.                      * --    "OpenOffice.org's Documentation of the Microsoft
  307.                      *         Excel File Format"
  308.                      **/
  309.                     $spos = $pos + 4;
  310.                     $limitpos = $spos + $length;
  311.                     $uniqueStrings = $this->_GetInt4d($this->_data, $spos + 4);
  312.                     $spos += 8;
  313.                     for ($i = 0; $i < $uniqueStrings; $i++) {
  314.                         // Read in the number of characters
  315.                         if ($spos == $limitpos) {
  316.                             $opcode = ord($this->_data[$spos]) | ord($this->_data[$spos + 1])<<8;
  317.                             $conlength = ord($this->_data[$spos + 2]) | ord($this->_data[$spos + 3])<<8;
  318.                             if ($opcode != 0x3c) {
  319.                                 return -1;
  320.                             }
  321.                             $spos += 4;
  322.                             $limitpos = $spos + $conlength;
  323.                         }
  324.                         $numChars = ord($this->_data[$spos]) | (ord($this->_data[$spos + 1]) << 8);
  325.                         $spos += 2;
  326.                         $optionFlags = ord($this->_data[$spos]);
  327.                         $spos++;
  328.                         $asciiEncoding = (($optionFlags & 0x01) == 0) ;
  329.                         $extendedString = ( ($optionFlags & 0x04) != 0);
  330.                         // See if string contains formatting information
  331.                         $richString = ( ($optionFlags & 0x08) != 0);
  332.                         if ($richString) { // Read in the crun
  333.                             $formattingRuns = ord($this->_data[$spos]) | (ord($this->_data[$spos + 1]) << 8);
  334.                             $spos += 2;
  335.                         }
  336.                         if ($extendedString) { // Read in cchExtRst
  337.                             $extendedRunLength = $this->_GetInt4d($this->_data, $spos);
  338.                             $spos += 4;
  339.                         }
  340.                         $len = ($asciiEncoding)? $numChars : $numChars*2;
  341.                         if ($spos + $len < $limitpos) {
  342.                             $retstr = substr($this->_data, $spos, $len);
  343.                             $spos += $len;
  344.  
  345.                         } else { // found countinue
  346.                             $retstr = substr($this->_data, $spos, $limitpos - $spos);
  347.                             $bytesRead = $limitpos - $spos;
  348.                             $charsLeft = $numChars - (($asciiEncoding) ? $bytesRead : ($bytesRead / 2));
  349.                             $spos = $limitpos;
  350.                             while ($charsLeft > 0){
  351.                                 $opcode = ord($this->_data[$spos]) | ord($this->_data[$spos + 1]) << 8;
  352.                                 $conlength = ord($this->_data[$spos + 2]) | ord($this->_data[$spos + 3]) << 8;
  353.                                 if ($opcode != 0x3c) {
  354.                                     return -1;
  355.                                 }
  356.                                 $spos += 4;
  357.                                 $limitpos = $spos + $conlength;
  358.                                 $option = ord($this->_data[$spos]);
  359.                                 $spos += 1;
  360.                                 if ($asciiEncoding && ($option == 0)) {
  361.                                     $len = min($charsLeft, $limitpos - $spos);
  362.                                     $retstr .= substr($this->_data, $spos, $len);
  363.                                     $charsLeft -= $len;
  364.                                     $asciiEncoding = true;
  365.  
  366.                                 } elseif (!$asciiEncoding && ($option != 0)) {
  367.                                     $len = min($charsLeft * 2, $limitpos - $spos);
  368.                                     $retstr .= substr($this->_data, $spos, $len);
  369.                                     $charsLeft -= $len/2;
  370.                                     $asciiEncoding = false;
  371.  
  372.                                 } elseif (!$asciiEncoding && ($option == 0)) {
  373.                                     // Bummer - the string starts off as Unicode,
  374.                                     // but after the continuation it is in
